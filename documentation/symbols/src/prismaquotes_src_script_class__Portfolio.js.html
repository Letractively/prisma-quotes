1 /**
  2  * @constructor
  3  * @param {string} name
  4  * @param {number} index
  5  * @param {boolean} sync
  6  */
  7 function Portfolio(name, index, visible) {
  8 	/**
  9 	 * The name of the portfolio
 10 	 * @type {string}
 11 	 */
 12   this.name = name;
 13 
 14 	/**
 15 	 * The portfolio's index (soon to be deprecated)
 16 	 * @type {number}
 17 	 */
 18   this.index = index;
 19 	
 20 	/**
 21 	 * An array of quotes included in the portfolio
 22 	 * @type {Array<Quote>}
 23 	 */
 24   this.quotes = [];
 25 	
 26 	/**
 27 	 * 
 28 	 * @type {Object}
 29 	 */
 30   this.$table;
 31 	
 32 	/**
 33 	 * Whether the portfolio information will be visible in a table generated by
 34 	 * a content script
 35 	 * @type {boolean}
 36 	 */
 37 	this.visible = (visible === undefined ? true : visible);
 38 }

 40 /**
 41  * Adds a quote to the portfolio
 42  * @param {Quote} quote
 43  * @returns The current object
 44  * @type Portfolio
 45  */
 46 Portfolio.prototype.add = function(quote) {
 47   this.quotes.push(quote);
 48 	return this;
 49 };

 51 /**
 52  * Creates a table for editing quote information on the options page
 53  * @param {object} $node A jQuery-wrapped node in which to place the table
 54  */
 55 Portfolio.prototype.editTable = function($node) {
 56 	var $div = $('<div>');
 57   var $table = $('<table>');
 58   var $thead = $('<thead>');
 59   var $tbody = $('<tbody>');
 60   var headers = ['index', 'secid', 'source', 'count', 'pricePaid', 'decimals', 'trigger'];
 61   var $theadTr = $('<tr>');
 62   
 63 	$.each(headers, function(index, header){
 64     $theadTr.append($('<th>').text(Localizer.translate(header)));
 65 	});
 66   $thead.append($theadTr);
 67 
 68   for (var i = 0; i < this.quotes.length; i++) {
 69     $tbody.append(this.addRow(this.quotes[i], i));
 70   }
 71   $tbody.append(this.addRow(new Quote('', 'bloomberg', 0, 0, 2, ''), i));
 72   
 73   $node.append($('<div>').append($('<input>').attr({
 74 			placeholder: Localizer.translate('newPortfolio'),
 75 			type: 'text'
 76 		}).change(function(){
 77 			var $parentDiv = $(this).parent();
 78 			var $nextDiv = $parentDiv.next();
 79 			if ($(this).val() === '') {
 80 				if ($nextDiv.size() !== 0 && $('tbody tr', $parentDiv).size() === 1) {
 81 					$parentDiv.remove();
 82 				}
 83 			} else {
 84 				if ($nextDiv.size() === 0) {
 85 					var p = new Portfolio('', 5);
 86 					p.editTable($node);
 87 				}				
 88 			}
 89 		}).val(this.name))
 90 		.append($('<button>').addClass('delete').click(function(){
 91 			var $parentDiv = $(this).parent();
 92 			if ($parentDiv.next().size() !== 0) {
 93 				if (confirm(Localizer.translate('deletePortfolio'))) {
 94 					$parentDiv.remove();
 95 				}
 96 			}
 97 		})).append($('<input>').attr({
 98 			checked: this.visible,
 99 			id: this.name + '_visible',
100 			type: 'checkbox'
101 		})).append($('<label>').attr('for', this.name + '_visible')
102 			.text(Localizer.translate('makeVisible')))
103 		.append($table.append($thead).append($tbody)));
104 
105   this.$table = $table;
106 	return this;
107 };

109 /**
110  * Exports the essential information in the object in JSON format
111  * @return A JSON representation
112  */
113 Portfolio.prototype.exportJSON = function(){
114 	var temp = [];
115 	$.each(this.quotes, function(index, quote){
116 		temp.push(quote.exportJSON());
117 	});
118 	return {
119 		name: this.name,
120 		index: this.index,
121 		visible: this.visible,
122 		quotes: temp
123 	};
124 };

126 /**
127  * Imports a JSON object to determine the portfolio's quotes
128  * @param {Array.<Object>} quotes A list of quotes to import, in JSON format
129  * @return {Portfolio} The current object
130  */
131 Portfolio.prototype.importJSON = function(quotes){
132 	$(quotes, function(index, quote){
133 		this.add(new Quote(quote.secid, quote.source, quote.count, quote.pricePaid,
134 			quote.decimals, quote.trigger));
135 	});
136 	return this;
137 };

139 /**
140  * Creates a row for the options page based on quote information
141  * @param {Object} quote The quote used to supply default values to the cells
142  * created
143  * @param {number} index The value to show in the index column
144  * @return
145  * @type Object
146  */
147 Portfolio.prototype.addRow = function(quote, index) {
148 	var valueCell = function(value, maxWidth){
149 		return $('<td>').append($('<input>').addClass('value').attr('type', 'text')
150 			.css('max-width', (maxWidth !== undefined ? maxWidth : 'auto')).val(value));
151 	}
152   var $tr = $('<tr>');
153 
154 	// # column
155   $tr.append($('<td>').text(index).addClass('value'));
156 
157   var parentScope = this;
158   $tr.append(	$('<td>').append($('<input>').attr('type', 'text').val(quote.secid).blur(function(e){
159 		var $this = $(this);
160 		if ($this.val() === '') {
161 			// If security ID is blank and we're not on the last row of the table,
162 			// delete the current row and select the following one
163 			var $tr = $this.closest('tr');
164 			var $trNext = $tr.next('tr');
165 			if ($trNext.size() > 0) {
166 				$trNext.find('input:first').focus();
167 				$tr.remove();
168 			}
169 		} else {
170 			if ($this.closest('tr').next('tr').size() === 0) {
171 				$this.closest('tbody').append(parentScope.addRow(new Quote('', 'bloomberg', 0, 0, 2, ''), index + 1))
172 			}
173 		}
174 		var $lastTr = $(this).closest('tr').siblings('tr:last');
175 	})));
176   
177 	// Source column
178 	$tr.append(Sources.selection(this.index + '_' + index, quote.source)); //$(Sources.selection(quote.source)));
179 	
180 	// Value columns
181 	$tr.append(valueCell(quote.count))
182   	.append(valueCell(quote.pricePaid))
183 		.append(valueCell(quote.decimals))
184 		.append(valueCell(quote.trigger, '20em'))
185 		.append($('<td>').append($('<button>').addClass('delete').click(function(){
186 			if ($tr.next().size() !== 0) {
187 				$tr.remove();
188 			}
189 		})));
190   
191   return $tr;
192 };
