  8 /**
  9  * This singleton contains all the methods used to sync local storage
 10  * information within the Other Bookmarks folder
 11  * @constructor
 12  */
 13 var Sync = (function(){
 14 	/**
 15 	 * The name of the bookmark
 16 	 * @const
 17 	 * @type string
 18 	 */
 19 	var BOOKMARK_NAME = 'Prisma Quotes Data';
 20 	
 21 	/**
 22 	 * The base URL for the bookmark
 23 	 * @const
 24 	 * @type string
 25 	 */
 26 	var BOOKMARK_URL = 'http://prismaquotes/?data=';
 27 	
 28 	/**
 29 	 * The main bookmark tree index
 30 	 * @const
 31 	 * @type number
 32 	 */
 33 	var MAIN_TREE_ID = 0;
 34 	
 35 	/**
 36 	 * Within the main tree, the Other Bookmarks folder is always given an index
 37 	 * of 1
 38 	 * @const
 39 	 * @type number
 40 	 */
 41 	var OTHER_BOOKMARKS_ID = 1;
 42 	
 43 	/**
 44 	 * A BookmarkTreeNode object that contains information on the bookmark node
 45 	 * being used for syncing
 46 	 * @type object
 47 	 */
 48 	var syncNode;
 49 	
 50 	/**
 51 	 * Whether currently syncing, to avoid event listeners from triggering tons
 52 	 * of events
 53 	 * @type boolean
 54 	 */
 55 	var syncing = false;
 56 	
 57 	/**
 58 	 * Determines the bookmark ID to sync with, creating one if necessary
 59 	 * @param {function} callback
 60 	 * @return The current object
 61 	 * @type Sync
 62 	 */
 63 	function getSyncId(callback){
 64 	  chrome.bookmarks.getTree(function(tree) {
 65 	    chrome.bookmarks.search(BOOKMARK_NAME, function(node) {
 66 	      if (node.length === 0) {
 67 					// Create a bookmark if one doesn't already exist
 68 					syncing = true;
 69 	        chrome.bookmarks.create({
 70 	          parentId: tree[MAIN_TREE_ID].children[OTHER_BOOKMARKS_ID].id,
 71 	          title: BOOKMARK_NAME,
 72 	          url: BOOKMARK_URL
 73 	        }, function(result) {
 74 						syncNode = result;
 75 						Sync.save();
 76 						syncing = false;
 77 						callback();
 78 					});
 79 	      } else {
 80 					syncNode = node[0];
 81 					callback();
 82 	      }
 83 	    });
 84 		});
 85 		return this;
 86 	};
 87 
 88 	/**
 89 	 * Handles bookmark change event
 90 	 * @param {number} id The ID of the bookmark affected
 91 	 * @param {Object.<string>} changeInfo The title and optional URL of what has
 92 	 * changed
 93 	 */
 94 	function bookmarkChanged(id, changeInfo){
 95 		if (syncNode.id === id) {
 96 			Sync.load();
 97 		}
 98 	};
 99 	
100 	/**
101 	 * Handles bookmark creation event
102 	 * @param {number} id The ID of the bookmark affected
103 	 * @param {BookmarkTreeNode} bookmark The bookmark created
104 	 */
105 	function bookmarkCreated(id, bookmark){
106 		/*
107 		 * Probably nothing.
108 		 * The only creation event we should worry about is when bookmarks are
109 		 * after import, and that should fire the onImported event.
110 		 */
111 	};
112 	
113 	/**
114 	 * Handles bookmark import event
115 	 */
116 	function bookmarkImported(){
117 		chrome.bookmarks.getTree(function(tree){
118 			chrome.bookmarks.search(BOOKMARK_NAME, function(results){
119 				// Determines which bookmark was most recently added if there are
120 				// duplicates
121 				var mostRecent = 0;
122 				$.each(results, function(index, result){
123 					if (result.dateAdded > mostRecent) {
124 						mostRecent = result.dateAdded;
125 						syncNode = result;
126 					}
127 				});
128 				
129 				// Removes earlier duplicates
130 				$.each(results, function(index, result){
131 					if (result.dateAdded !== mostRecent) {
132 						chrome.bookmarks.remove(result.id);
133 					}
134 				});
135 				Sync.load();
136 			});
137 		});
138 	};
139 
140 	/**
141 	 * Handles bookmark removal event
142 	 * @param {number} id The ID of the bookmark affected
143 	 * @param {Object.<(string||integer)>} The parentId and index of the bookmark
144 	 * removed
145 	 */
146 	function bookmarkRemoved(id, removeInfo){
147 		// Probably nothing necessary here
148 	};
149 
150 	return {
151 		/**
152 		 * Returns the ID associated with the sync bookmark node
153 		 * @return {number} The ID number for the bookmark we're syncing with
154 		 */
155 		id: function(){
156 			return syncNode.id;
157 		},
158 		
159 		/**
160 		 * Arms all of the events that will trigger a reload
161 		 * @return The current object
162 		 * @type Sync
163 		 */
164 		initialize: function(){
165 			// There's probably a slight performance hit or something from monitoring
166 			// changes in bookmarks, but this way we don't have to use a bunch of
167 			// callbacks when saving from the options screen
168 			getSyncId(function(){
169 				// Don't start listening for events until we have the right bookmark
170 				Sync.listen();
171 				
172 				// Probably overkill, but this is just to make sure we don't miss any
173 				// sync events prior to figuring out the bookmark node ID
174 				if (Settings.lastSync() < syncNode.dateAdded) {
175 					Sync.load();
176 				}
177 			});
178 			return this;
179 		},
180 
181 		/**
182 		 * Attaches and detaches event listeners
183 		 * @param {boolean} toggle Whether to add or remove listeners, defaulting
184 		 * to true
185 		 * @return The current object
186 		 * @type Sync
187 		 */
188 		listen: function(toggle){
189 			if (toggle === undefined || toggle) {
190 				chrome.bookmarks.onChanged.addListener(bookmarkChanged);
191 				chrome.bookmarks.onCreated.addListener(bookmarkCreated);
192 				chrome.bookmarks.onImportEnded.addListener(bookmarkImported);
193 				chrome.bookmarks.onRemoved.addListener(bookmarkRemoved);
194 			} else {
195 				chrome.bookmarks.onChanged.removeListener(bookmarkChanged);
196 				chrome.bookmarks.onCreated.removeListener(bookmarkCreated);
197 				chrome.bookmarks.onImportEnded.removeListener(bookmarkImported);
198 				chrome.bookmarks.onRemoved.removeListener(bookmarkRemoved);
199 			}
200 			return this;
201 		},
202 
203 		/**
204 		 * Loads information from bookmark
205 		 * @return {Sync} The current object
206 		 */
207 		load: function(){
208 			// Only run if not already syncing
209 			if (Settings.sync() && !syncing) {
210 				try {
211 					// The old node gets deleted, so we have to find the new one that's
212 					// replaced it
213 					getSyncId(function(){
214 						var data = JSON.parse(unescape(syncNode.url.replace(BOOKMARK_URL, '')));
215 						$.each(data, function(key, value){
216 							localStorage[key] = (key === 'sync' ? 'true' : value);
217 						});
218 						window.location.reload();
219 					});
220 				} 
221 				catch (e) {
222 					// error handling
223 				}
224 			}
225 			return this;
226 		},
227 
228 		overwrite: function(){
229 			var data = JSON.parse(unescape(syncNode.url.replace(BOOKMARK_URL, '')));
230 			alert(JSON.parse(data.lastSync) + ', ' + Settings.lastSync() + ', ' + (JSON.parse(data.lastSync) > Settings.lastSync));
231 			if (JSON.parse(data.lastSync) > Settings.lastSync()) {
232 				for (var key in data) {
233 					localStorage[key] = data[key];
234 				}
235 				alert('reload time');
236 				var d = new Date();
237 				Settings.lastSync(d.getTime());
238 				window.location.reload();
239 			}			
240 		},
241 
242 		/**
243 		 * Saves local storage to bookmark
244 		 * @return The current object
245 		 * @type Sync
246 		 */
247 		save: function(){
248 			// Saves regardless of whether sync is on or not. This is to avoid
249 			// overwriting everything with some old bookmark if sync gets turned on.
250 			// This is probably not ever an issue, but I'll get my bearings first.
251 			var d = new Date();
252 			syncing = true;
253 			Settings.lastSync(d.getTime());
254       chrome.bookmarks.update(syncNode.id, {
255         url: BOOKMARK_URL + JSON.stringify(localStorage)
256       }, function(result){
257 				syncing = false;
258 			});
259 			return this;
260 		}
261 	}
262 })();
263 
264 Sync.initialize();
