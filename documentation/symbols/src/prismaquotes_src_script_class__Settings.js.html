 16 	function getVersion(){
 17 		var xhr = new XMLHttpRequest()
 18 		xhr.open('GET', chrome.extension.getURL('manifest.json'), false);
 19 		xhr.send('');
 20 		return JSON.parse(xhr.responseText).version;
 21 	};

 64 	return {
 65 		/**
 66 		 * Loads a value from local storage
 67 		 * @param {string} key The name of the parameter
 68 		 * @return {*} The value requested
 69 		 */
 70 		get: function(key){
 71 			return JSON.parse(localStorage[key]);
 72 		},
 73 		
 74 		/**
 75 		 * Saves a value to local storage
 76 		 * @param {string} key The name of the parameter
 77 		 * @param {Object} value The value to store
 78 		 * @return {Settings} The current object
 79 		 */
 80 		set: function(key, value){
 81 			localStorage[key] = JSON.stringify(value);
 82 			return this;
 83 		},
 84 
 85 		/**
 86 		 * Exports the essential information in the object in JSON format
 87 		 * @return A JSON representation
 88 		 * @type Object
 89 		 */
 90 		exportJSON: function(){
 91 			var temp = {};
 92 			for (var key in localStorage) {
 93 				try {
 94 					// If uncaught, JSON.parse errors can prevent the entire page from loading
 95 					temp[key] = JSON.parse(localStorage[key]);
 96 				} catch(e) {
 97 					console.log('key ' + key + ' returns an unexpected token ILLEGAL error from JSON.parse')
 98 				}
 99 			}
100 			return temp;
101 		},
102 
103 		/**
104 		 * Gets or sets the behavior data in local storage
105 		 * @param {string|Object=} args A string indicated the key to set or
106 		 * retrieve, or a hash containing the list of keys and their values to set
107 		 * @param {string=} value Optional value
108 		 * @return The value requested or the current object
109 		 * @type {string|Settings}
110 		 */
111 		behaviors: function(args, value){
112 			switch (typeof args) {
113 				case 'undefined':
114 					return this.get('behaviors');
115 					break;
116 				case 'string':
117 					var temp = this.get('behaviors');
118 					if (value !== undefined) {
119 						temp[args] = value;
120 						return this.set('behaviors', temp);
121 					} else {
122 						return temp[args];
123 					}
124 					break;
125 				case 'object':
126 					var temp = this.get('behaviors');
127 					$.each(args, function(key, value){
128 						temp[key] = value;
129 					})
130 					return this.set('behaviors', temp);
131 					break;
132 				default:
133 					return this.get('behaviors');
134 					break;
135 			}
136 		},

140 		 * @param {Object} value Column data
141 		 * @returns The value requested or the current object

143 		columns: function(value){
144 			return (value !== undefined) ? this.set('columns', value) : this.get('columns');
145 		},

158 		 * @param {boolean} value True or false
159 		 * @return The value requested or the current object
160 		 * @type {boolean|Settings}
161 		 */
162 		inject: function(value){
163 			return (value !== undefined) ? this.set('inject', value) : this.get('inject');			
164 		},
165 
166 		/**
167 		 * Gets or sets whether the injected popup is visible
168 		 * @param {boolean} value True or false
169 		 * @return The value requested or the current object
170 		 * @type {boolean|Settings}
171 		 */
172 		injectVisible: function(value){
173 			return (value !== undefined) ? this.set('injectVisible', value) : this.get('injectVisible');			
174 		},
175 		
176 		/**
177 		 * Verifies that all required settings exist in local storage and that they
178 		 * have the correct type, resetting them if not
179 		 * @return The current object
180 		 * @type Settings
181 		 */
182 		load: function(){
183 			var parentScope = this;
184 			$.each(DEFAULTS, function(key, defaultValue){
185 				try {
186 					if (key in localStorage) {
187 						var temp = JSON.parse(localStorage[key])
188 						
189 						// Make sure columns work fine in version 2
190 						if (key === 'columns' && temp !== undefined &&
191 							  (typeof temp[0]).toLowerCase() === 'string') {
192 							parentScope.set(key, defaultValue);
193 						}
194 						
195 						// Test whether type is the same between stored values and defaults 
196 						if (typeof temp !== typeof defaultValue || temp === undefined) {
197 							parentScope.set(key, defaultValue)
198 						}
199 					} else {
200 						parentScope.set(key, defaultValue);
201 					}
202 				} catch(e) {
203 					parentScope.set(key, defaultValue);
204 				}
205 			});
206 			return this;
207 		},

234 		},

236 		/**
237 		 * Resets the key or keys provided, or everything if no argument is passed
238 		 * @param {string|Array} key A key, an array of keys or nothing
239 		 * @return The current object
240 		 * @type Settings
241 		 */
242 		reset: function(keys){
243 			if (keys !== undefined) {
244 				$.each($.isArray(keys) ? keys : [keys], function(index, key){
245 					try {
246 						if (key in DEFAULTS) {
247 							Settings.set(key, DEFAULTS[key]);
248 						}
249 					} catch(e) {
250 						// error handling
251 					}
252 				});
253 			} else {
254 				// If no key is defined specifically, prompt user to reset everything
255 				$.each(DEFAULTS, function(key, defaultValue){
256 					Settings.set(key, defaultValue);
257 				});
258 			}
259 			return this;
260 		},
