 19 	/**
 20 	 * Determines if there are commas left to place in an unformatted number
 21 	 * @const
 22 	 * @type regex
 23 	 */
 24 	var DECIMAL_REGEX = /(\d+)(\d{3})/;
 25 
 26 	/**
 27 	 * Adds commas to a number
 28 	 * @param {string} num An unformatted number
 29 	 * @return The formatted number
 30 	 * @type string
 31 	 */
 32 	function fix(num){
 33 		var unit = 'KB';
 34 		if (num > 1048576) {
 35 			num = num / 1048576
 36 			unit = 'GB';
 37 		} else if (num > 1024) {
 38 			num = num / 1024
 39 			unit = 'MB'
 40 		}
 41 		var temp = num.toFixed(1).split('.');
 42 
 43 		// Repeat as long as there are more than three digits in a row
 44 		while (DECIMAL_REGEX.test(temp[0])) {
 45 			temp[0] = temp[0].replace(DECIMAL_REGEX, '$1,$2');
 46 		}
 47 		return temp.join('.') + ' ' + unit;
 48 	};

 54 		networkUsage: function(){
 55 			var sums = [0, 0, 0, 0];
 56 			var selectedSources = $('#portfolioEdit tr:not(:last) input[type=radio]:checked');
 57 			$('#networkUsage tbody tr').each(function(){
 58 				var $tr = $(this);
 59 				var $tds = $('td', $tr);
 60 				var source = $tds.eq(0).data('i18n');
 61 				var count = selectedSources.filter(function(){return $(this).val()===source}).size();
 62 				sums[0] += count;
 63 				$tds.eq(1).text(count);
 64 				var cost = parseFloat($tds.eq(2).text().replace(' kbps', ''));
 65 				var frequency = 3600000 / parseFloat($('#frequency').val());
 66 				var perHour = count * cost * frequency;
 67 				sums[1] += perHour;
 68 				$tds.eq(3).text(fix(perHour));
 69 				var perDay = perHour * 24;
 70 				sums[2] += perDay;
 71 				$tds.eq(4).text(fix(perDay));
 72 				var perMonth = perDay * 30;
 73 				sums[3] += perMonth;
 74 				$tds.eq(5).text(fix(perMonth));
 75 			});
 76 			var $tfootTds = $('#networkUsage tfoot td');
 77 			$tfootTds.eq(1).text(sums[0]);
 78 			$tfootTds.eq(3).text(fix(sums[1]));
 79 			$tfootTds.eq(4).text(fix(sums[2]));
 80 			$tfootTds.eq(5).text(fix(sums[3]));
 81 		},

130 		/**
131 		 * Parses the columns information on the options page
132 		 * @return {Object} A JSON representation of columns information to save
133 		 * @see Settings#columns
134 		 */
135 		parseColumns: function(){
136 			var columns = [];
137 			$('#metricsEdit li:not(:last)').each(function(){
138 				try {
139 					columns.push({
140 						metric: $(this).children('select').val(),
141 						inject: $(this).children('input').attr('checked')

143 				} catch(e) {
144 					// error handling
145 				}
146 			});
147 			return columns;
148 		},

150 		/**
151 		 * Parses the portfolio information on the options page
152 		 * @return {Object} A JSON representation of portfolio information to save
153 		 * @see Settings#portfolios
154 		 */
155 		parsePortfolio: function(){
156 			var investor = {};
157 			$('#portfolioEdit div:not(:last)').each(function(){
158 				var $div = $(this);
159 				var key = $div.children('input[type=text]:first').val();
160 				var portfolio = {
161 					visible: $div.children('input[type=checkbox]:first').attr('checked'),
162 					quotes: []
163 				};
164 				$('tbody tr:not(:last)', $div).each(function(){
165 					var $inputs = $('input[type=text]', $(this));
166 					var $checked = $('input[type=radio]:checked', $(this));
167 					portfolio.quotes.push({
168 						secid: $inputs.eq(0).val(),
169 						source: ($checked.size() > 0 ? $checked.val() : 'bloomberg'),
170 						count: parseFloat($inputs.eq(1).val()),
171 						pricePaid: parseFloat($inputs.eq(2).val()),
172 						decimals: parseFloat($inputs.eq(3).val()),
173 						trigger: $inputs.eq(4).val()
174 					});
175 				})
176 				investor[key] = portfolio;
177 			});
178 			return investor;
179 		},

181 		/**
182 		 * Saves all of the options on the page
183 		 * @returns The current object
184 		 * @type Options
185 		 */
186 		save: function(){
187 			try {
188 				var prevSync = bg.Settings.sync();
189 				bg.Settings.behaviors({
190 						inject: $('#behaviorInject').val(),
191 						popup: $('#behaviorPopup').val()
192 					})
193 					.columns(this.parseColumns())
194 					.frequency(parseFloat($('#frequency').val()))
195 					.inject($('#inject').attr('checked'))
196 					.language($('#language').val())
197 					.portfolios(this.parsePortfolio())
198 					.sync($('#sync').attr('checked'));
199 				
200 				// Only sync if sync is turned on and we weren't syncing previously.
201 				// This prevents the nasty situation where you overwrite the existing
202 				// bookmark with stuff like the demo information
203 				if (bg.Settings.sync()) {
204 					if (prevSync) {
205 						bg.Sync.save();						
206 					} else {
207 						if (confirm(bg.Localizer.translate('syncNow'))) {
208 							bg.Sync.load();
209 						}						
210 					}
211 				}
212 				bg.Localizer.load(function(){
213 					alert(bg.Localizer.translate('saveAlert'));
214 					bg.Background.reload();
215 					window.location.reload();
216 				});
217 			} 
218 			catch (e) {
219 				alert("Error saving!");
220 			}
221 			return this;
222 		},

251 			
252 			bg.Localizer.localize($(document));
253 			this.networkUsage();
254 			return this;
255 		}
