  8 /**
  9  * This class represents an individual investor, who may have a collection of
 10  * portfolios, and within those a collection of quotes
 11  * @constructor
 12  */
 13 var Investor = (function(){
 14   var portfolios = [];
 15   var currentIndex = 0;
 16   var timeout;
 17 	
 18 	return {
 19 		/**
 20 		 * Adds a portfolio
 21 		 * @param {Portfolio} portfolio The portfolio to be added
 22 		 * @returns The current object
 23 		 * @type Investor
 24 		 */
 25 		add: function(portfolio){
 26 			portfolios.push(portfolio);
 27 			return this;
 28 		},
 29 
 30 		/**
 31 		 * Clears all portfolio-related data
 32 		 * @return {Investor} The current object
 33 		 */
 34 		clear: function(){
 35 		  portfolios = [new Portfolio("Portfolio", 0)];
 36 		  Investor.save();
 37 			window.location.reload();
 38 			return this;
 39 		},
 40 		
 41 		/**
 42 		 * Clears quote history from Bloomberg cookies, which has a tendency to
 43 		 * return errors from the server
 44 		 * @return {Investor} The current object
 45 		 */
 46 		clearCookies: function(){
 47 			chrome.cookies.remove({
 48 				url: 'http://www.bloomberg.com/apps',
 49 				name: 'bbquotehistory'
 50 			});
 51 			return this;
 52 		},
 53 		
 54 		/**
 55 		 * Retrieves the number of portfolios assigned to the Investor object
 56 		 * @returns The number of portfolios
 57 		 * @type Number
 58 		 */
 59 		count: function(){
 60 			return portfolios.length;
 61 		},
 62 
 63 		/**
 64 		 * Retrieves the current portfolio
 65 		 * @returns The current portfolio
 66 		 * @type Portfolio
 67 		 */
 68 		current: function(){
 69 			var iter = 0;
 70 			while (portfolios[currentIndex] == undefined && currentIndex > 0 && iter < 50) {
 71 				currentIndex--;
 72 				iter++;
 73 			}
 74 			return portfolios[currentIndex];
 75 		},
 76 		
 77 		/**
 78 		 * Removes the trigger condition from all related quotes
 79 		 * @param {string} secid The security ID shown in the notification window
 80 		 * @param {string} trigger The specific trigger condition to remove
 81 		 * @returns The current object
 82 		 * @type Investor
 83 		 */
 84 		disarm: function(secid, trigger){
 85 			$.each(portfolios, function(i, portfolio){
 86 				$.each(portfolio.quotes, function(j, quote){
 87 					if (quote.secid === secid) {
 88 						quote.trigger = quote.trigger.replace(trigger, '').replace(/^\,|\,$/g, "").replace(/\,\,/g, ",");
 89 						quote.triggered = false;
 90 						quote.checkTrigger();
 91 					}
 92 				});
 93 			});
 94 			return this;
 95 		},		
 96 		
 97 		/**
 98 		 * Drew table in previous versions
 99 		 * @param {Object} contextNode
100 		 * @deprecated
101 		 */
102 		drawTable: function(contextNode){
103 			Investor.current().drawTable(contextNode);
104 		},
105 		
106 		editTable: function($node){
107 			$.each(portfolios, function(index, portfolio){
108 				portfolio.editTable($node);
109 			});
110 			var p = new Portfolio('', portfolios.length)
111 			p.editTable($node);
112 		},
113 		
114 		/**
115 		 * Exports the essential information in the object in JSON format
116 		 * @return A JSON representation
117 		 */
118 		exportJSON: function(){
119 			var temp = [];
120 			$.each(portfolios, function(index, portfolio){
121 				temp.push(portfolio.exportJSON());
122 			});
123 			return temp;
124 		},
125 		
126 		/**
127 		 * Imports a JSON object to construct the portfolios and quotes
128 		 * @param {Array.<Object>} A JSON representation of investor information
129 		 * @return {Investor} The current object
130 		 */
131 		importJSON: function(investorJSON){
132 			portfolios = [];
133 			$.each(investorJSON, function(index, portfolioJSON){
134 				var portfolio = new Portfolio(portfolioJSON.name, portfolioJSON.index,
135 					portfolioJSON.visible);
136 				portfolio.importJSON(portfolioJSON.quotes);
137 				portfolios.push(portfolio);
138 			});
139 			return this;
140 		},
141 		
142 		/**
143 		 * Loads the settings for portfolios and quotes into the Investor object
144 		 * @returns The current object
145 		 * @type Investor
146 		 */
147 		load: function(){
148 			var parentScope = this;
149 			var i = 0;
150 			$.each(Settings.portfolios(), function(name, portfolio){
151 				if (portfolio.visible === undefined) {
152 					portfolio = {
153 						quotes: portfolio,
154 						visible: true
155 					};
156 				}
157 				
158 				var p = new Portfolio(name, i++, portfolio.visible);
159 				$.each(portfolio.quotes, function(key, data){
160 					var q = new Quote(data.secid, data.source, data.count,
161 						data.pricePaid, data.decimals, data.trigger);
162 					p.add(q);
163 				});
164 				parentScope.add(p);
165 			});
166 			this.loadTickers();
167 			setInterval(this.loadTickers, Settings.frequency());
168 			return this;
169 		},
170 
171 		/**
172 		 * Calls the loadTicker method in each {@link Quote} object
173 		 * @return {Investor} The current object
174 		 * @see Quote#loadTicker
175 		 */
176 		loadTickers: function(){
177 			Investor.clearCookies();
178 			
179 			$.each(portfolios, function(i, portfolio){
180 				$.each(portfolio.quotes, function(j, quote){
181 					quote.loadTicker();
182 				});
183 			});
184 			return this;
185 		},
186 		
187 		/**
188 		 * Resets the portfolios in the Investor object to their demo values
189 		 * @return The current object
190 		 * @type Investor
191 		 */
192 		reset: function(){
193 			Settings.reset('portfolios');
194 			window.location.reload();
195 			return this;
196 		},
197 		
198 		/**
199 		 * Removes the portfolio assigned to a specific index
200 		 * @param {Number} index
201 		 * @returns The current object
202 		 * @type Investor
203 		 */
204 		remove: function(index){
205 			portfolios.splice(index, 1);
206 			return this;
207 		},
208 		
209 		/**
210 		 * Saves the portfolios and quotes assigned to the Investor object to local
211 		 * storage
212 		 * @returns The current object
213 		 * @type Investor
214 		 */
215 		save: function(){
216 		  var saveJSON = {};
217 			$.each(portfolios, function(index, portfolio){
218 		    saveJSON[portfolio.name] = {
219 					visible: portfolio.visible,
220 					quotes: []
221 				};
222 				$.each(portfolio.quotes, function(index, quote){
223 		      saveJSON[portfolio.name].quotes.push({
224 		        secid: quote.secid,
225 		        source: quote.source,
226 		        count: quote.count,
227 		        pricePaid: quote.pricePaid,
228 		        decimals: quote.decimals,
229 		        trigger: quote.trigger
230 		      });
231 				});
232 			});
233 		  Settings.portfolios(saveJSON);
234 			return this;
235 		}
236 	}
237 })();
238 
239 Investor.load();
