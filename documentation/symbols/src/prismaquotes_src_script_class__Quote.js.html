  8 /**
  9  * This is the object which represents an individual security quote
 10  * @constructor
 11  * @param {string} secid The security ID
 12  * @param {string} source The pricing source (Bloomberg, Google, Yahoo!, Quote.com)
 13  * @param {number} count The number of shares owned
 14  * @param {number} pricePaid The price paid for each of the shares owned
 15  * @param {number} decimals The number of decimals to include in the price
 16  * @param {string} trigger The conditions for alerting the user
 17  * @param {boolean} isTotal Currently unused
 18  */
 19 function Quote(secid, source, count, pricePaid, decimals, trigger, isTotal) {
 20   this.secid = secid;
 21   this.source = source;
 22   this.count = count;
 23   this.pricePaid = pricePaid;
 24   this.title = "";
 25   this.value = 0;
 26   this.change = 0;
 27 	this.others = {};
 28   this.decimals = decimals;
 29   this.trigger = trigger;
 30   this.triggered = false;
 31   this.tr;
 32 }

 63 /**
 64  * Loads tickers via a $.ajax call to the source URL and parsing the output
 65  * with a regular expression
 66  * @returns The current object
 67  * @type Quote
 68  */
 69 Quote.prototype.loadTicker = function() {
 70   var regex = Sources.get(this.source).regex;
 71   var parentScope = this;
 72 
 73 	$.ajax({
 74 		type: 'get',
 75 		url: Sources.get(this.source).url + this.secid,
 76 		cache: false,
 77 		dataType: 'text',
 78 		success: function(vals){
 79 			var titleMatches = regex.title.exec(vals);
 80 			if (titleMatches) {
 81 				parentScope.title = titleMatches[1];
 82 			}
 83 			
 84 			var valueMatches = regex.value.exec(vals);
 85 			if (valueMatches) {
 86 				parentScope.value = parseFloat(valueMatches[1].replace(/,/g, ''));
 87 			}
 88 	
 89 			var changeMatches = regex.change.exec(vals);
 90 			if (changeMatches) {
 91 				if (parentScope.source == "yahoo" &&
 92 				changeMatches[0].indexOf("price-change-down") !== -1) {
 93 					parentScope.change = -parseFloat(changeMatches[1].replace(/,/g, ''));
 94 				}
 95 				else {
 96 					parentScope.change = parseFloat(changeMatches[1].replace(/,/g, ''));
 97 				}
 98 				
 99 				// Set change to 0 if it's invalid
100 				if (isNaN(Number(parentScope.change))) {
101 					parentScope.change = 0;
102 				}
103 			}
104 			
105 			$.each(regex.others, function(key, re){
106 				var othersInUse = Metrics.using();
107 				try {
108 					// To minimize the number of regular expression run, only test
109 					// other metrics if they're currently designated for use
110 					if (key in othersInUse) {
111 						var matches = re.exec(vals);
112 						
113 						if (Metrics.type(key) !== 'text') {
114 							// Convert places, such as for volume
115 							var places = 0;
116 							var match = matches[1].replace(/,/g, '');
117 							switch (match.substr(-1).toLowerCase()) {
118 								case 'k':
119 									places = 1000;
120 									break;
121 								case 'm':
122 									places = 1000000;
123 									break;
124 								case 'b':
125 									places = 1000000000;
126 									break;
127 								case 't':
128 									places = 1000000000000;
129 									break;
130 								default:
131 									break;
132 							}
133 							if (key === 'volume') {
134 								console.log(matches);
135 								console.log(match.substr(-1));
136 								console.log(parseFloat(match.substr(0, match.length - 1)));
137 							}
138 							if (places !== 0) {
139 								parentScope.others[key] = parseFloat(match.substr(0, match.length - 1)) * places;
140 							}
141 							else {
142 								parentScope.others[key] = parseFloat(match);
143 							}
144 						}
145 						else {
146 							parentScope.others[key] = matches[1];
147 						}
148 					}
149 				} catch(e) {
150 					parentScope.others[key] = "";
151 				}
152 			});
153 
154       parentScope.checkTrigger();
155 		}
156 	});
157 	return this;
158 };

160 /**
161  * Checks whether a trigger condition has been met, and if so alerts the user
162  * @returns The current object
163  * @type Quote
164  */
165 Quote.prototype.checkTrigger = function() {
166   if (!this.triggered && this.trigger !== "") {
167     var temp = this.trigger.split(",");
168     for (var i = 0; i < temp.length; i++) {
169       var operator = temp[i].substring(0, 1) +
170 				(temp[i].substring(1, 2) === "=" ? "=" : "");
171       var limit = parseFloat(temp[i].replace(operator, ""));
172       var pastLimit = false;
173       var preposition = "";
174       
175       switch (temp[i].substring(0, 1) + (temp[i].substring(1, 2) === "=" ? "=" : "")) {
176         case ">":
177           preposition = "above";
178           pastLimit = (this.value > limit);
179           break;
180         case ">=":
181           preposition = "above";
182           pastLimit = (this.value >= limit);
183           break;
184         case "<=":
185           preposition = "below";
186           pastLimit = (this.value <= limit);
187           break;
188         case "<":
189           preposition = "below";
190           pastLimit = (this.value < limit);
191           break;
192         default:
193           break;
194       }
195 
196       if (pastLimit) {
197         var notification = webkitNotifications.createHTMLNotification(
198           chrome.extension.getURL("notification.html?secid=" + this.secid +
199             "&value=" + this.value.toFixed(this.decimals) +
200             "&limit=" + limit.toFixed(this.decimals) +
201             "&preposition=" + preposition +
202             "&trigger=" + temp[i]));
203         notification.show();
204         this.triggered = true;
205         break;
206       }
207     }
208   }
209 	return this;
210 };
