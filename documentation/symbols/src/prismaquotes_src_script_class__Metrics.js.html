  7 /**
  8  * This class contains the metrics available for use in returning values for
  9  * the security IDs chosen
 10  * @constructor
 11  */
 12 var Metrics = (function(){
 13   /**
 14    * A private object holding the metrics available for use
 15    * @type Object
 16    */
 17   var metrics = {};
 18   
 19   /**
 20    * A private array holding the metrics chosen for use
 21    * @type Object
 22    */
 23   var columns = [];
 24   
 25   /**
 26    * A private member containing a jQuery-wrapped DIV node
 27    * @type Object
 28    */
 29   var $div;
 30   
 31   /**
 32    * Creates a selection dropdown of metrics for the options page
 33    * @param {Object} column The name of the metric currently chosen for this
 34    * index
 35    * @return A jQuery-wrapped selection dropdown containing a list of metrics
 36    * @type Object
 37    */
 38   function selection(column, index){
 39 		var $li = $('<li>');
 40 		
 41     var $select = $('<select>').append($('<option>')).change(function(){
 42       var $this = $(this);
 43 			var $li = $this.parent('li');
 44 			var $nextLi = $li.next('li');
 45       if ($this.val() === '') {
 46         if ($nextLi.size() > 0) {
 47           $li.remove();
 48         }
 49       } else {
 50         if ($nextLi.size() === 0) {
 51           $li.after(selection(''));
 52         }
 53       }
 54     });
 55     
 56     $.each(metrics, function(key, value){
 57       $select.append($('<option>').val(key).text(Localizer.translate(key)));
 58     });
 59     $select.val(column.metric);
 60     return $li.append($select).append($('<input>').attr({
 61 			checked: column.inject,
 62 			id: 'metric_' + index,
 63 			type: 'checkbox'
 64 		})).append($('<label>').text(Localizer.translate('makeVisible'))
 65 			.attr('for', 'metric_' + index));
 66   }
 67   
 68   return {
 69     /**
 70      * Adds a new metric to the Metrics object
 71      * @param {string} label The label which will be used when localizing the
 72      * name of the parameter
 73      * @param {string} type The type of value (text, value, change, pct)
 74      * @param {function} func The function for evaluating the sum of all values
 75      * @param {function} numer The function for evaluating the numerator
 76      * @param {function} denom The function for evaluating the denominator
 77      * @returns The current object
 78      * @type Metrics
 79      */
 80     add: function(label, type, func, numer, denom, places){
 81       if (numer !== undefined && numer !== null) {
 82         metrics[label] = {
 83           type: type,
 84           func: func,
 85 					valid: Sources.valid(label),
 86           numer: numer,
 87           denom: denom,
 88 					places: (places !== undefined ? places : 1)
 89         };
 90       }
 91       else {
 92         metrics[label] = {
 93           type: type,
 94           func: func,
 95 					valid: Sources.valid(label),
 96 					places: (places !== undefined ? places : 1)
 97         };
 98       }
 99       return this;
100     },
101     
102     /**
103      * Evaluates a metric for the supplied context
104      * @param {string} label The metric's label
105      * @param {Object} context
106      * @returns
107      * @type number
108      */
109     calculate: function(label, context){
110       return metrics[label].func.call(context);
111     },
112     
113     /**
114      * Retrieves the metric assigned to a particular column
115      * @param {Object} index The index of the column
116      * @returns A metric and its associated functions
117      * @type Object
118      */
119     column: function(index){
120       return columns[index];
121     },
122     
123     /**
124      * Retrieves the number of metrics in use
125      * @returns The number of metrics in use
126      * @type number
127      */
128     count: function(){
129       return columns.length;
130     },
131     
132     /**
133      * Creates a dropdown of available for metrics for selection on the options
134      * page
135      * @param {Object} $node The jQuery-wrapped DOM node that will contain this
136      * method's output
137      * @returns The current object
138      * @type Metrics
139      */
140     dropdowns: function($node){
141       var $ul = $('<ul>').appendTo($node.empty());
142       
143       $.each(columns, function(index, column){
144         $ul.append(selection(column, index));
145       });
146       $ul.append(selection('', columns.length));
147       
148       $div = $node;
149       return this;
150     },
151     
152     /**
153      * Exports Metrics object to JSON format
154      * @return Blah
155      * @type Object
156      */
157     exportJSON: function(){
158       var temp = [];
159       $.each(columns, function(index, column){
160         var metric = metrics[column.metric];
161         metric.name = column.metric;
162 				metric.inject = column.inject;
163         temp.push(metric);
164       });
165       return temp;
166     },
167     
168     /**
169      * Retrieves a metric based on its corresponding label
170      * @param {string} label
171      * @returns A metric
172      * @type object
173      */
174     get: function(label){
175       return metrics[label];
176     },
177     
178     /**
179      * Loads the values for the metrics in use from local storage
180      * @returns The current object
181      * @type Metrics
182      */
183     load: function(){
184       try {
185         columns = Settings.columns();
186       } 
187       catch (e) {
188         // Metrics.reset();
189       }
190       return this;
191     },
192     
193     /**
194      * Resets the metrics in use to their defaults
195      * @returns The current object
196      * @type Metrics
197      * @see Settings#reset
198      */
199     reset: function(){
200 			Settings.reset('columns');
201       return this;
202     },
203     
204     /**
205      * Saves the metrics based on selections made on the options page
206      * @returns The current object
207      * @type Metrics
208      */
209     save: function(){
210       var $selects = $('select', $div);
211       var temp = [];
212       
213       $('select', div).each(function(){
214         if ($(this).val() !== "") {
215           temp.push($(this).val());
216         }
217       });
218       
219       Settings.columns(temp);
220       columns = temp;
221       return this;
222     },
223 		
224     /**
225      * Retrieves the type of a metric based on its corresponding label
226      * @param {string} label
227      * @return {string} The metric's type
228      * @type object
229      */
230 		type: function(label) {
231 			return metrics[label].type;
232 		},
233 		
234 		/**
235 		 * Returns a list of other metrics currently in use (i.e. excluding basic
236 		 * information such as price, change or the security's description)
237 		 * @return {Object.<string>} The names of metrics in use
238 		 */
239 		using: function(){
240 			var temp = {};
241 			$.each(Settings.columns(), function(index, column){
242 				switch (column.metric) {
243 					case 'secid':
244 					case 'title':
245 					case 'value':
246 					case 'change':
247 					case 'pctChange':
248 						break;
249 					default:
250 						temp[column.metric] = '';
251 						break;
252 				}
253 			});
254 			return temp;
255 		}
256   }
257 })();
