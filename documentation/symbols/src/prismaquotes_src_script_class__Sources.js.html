  8 /**
  9  * This is the Sources singleton, used for storing information related to quote
 10  * information sources such as Bloomberg, Google and Yahoo!
 11  * @constructor
 12  */
 13 var Sources = (function(){
 14 	/**
 15 	 * A list of the sources currently available
 16 	 * @type Object.<string, Object>
 17 	 */
 18 	var sources = {};
 19 	
 20 	/**
 21 	 * The context node in which to create a list of dropdowns
 22 	 * @type Node
 23 	 */
 24 	var div;
 25 	
 26 	return {
 27 		/**
 28 		 * Adds the information pertaining to a source
 29 		 * @param {string} key The key for the source
 30 		 * @param {string} url The URL containing quote information (less any
 31 		 * ticker symbol)
 32 		 * @param {string} src The source for the chart
 33 		 * @param {string} flashVars Any flashvars necessary to pass to the embed
 34 		 * code
 35 		 * @param {number} width The width of the chart
 36 		 * @param {number} height The height of the chart
 37 		 * @param {number} opacity The opacity of the chart's background (either
 38 		 * "transparent" or "opaque")
 39 		 * @param {RegExp} reTitle The regular expression for determining the title
 40 		 * of the security
 41 		 * @param {RegExp} reValue The regular expression for determining the value
 42 		 * of the security
 43 		 * @param {RegExp} reChange The regular expression for determining the
 44 		 * change in the value of the security
 45 		 * @param {boolean} plainOn Whether to show static charts in place of
 46 		 * interactive ones
 47 		 * @param {string} plainURL URL for a static chart
 48 		 * @param {number} plainWidth The width of the static chart
 49 		 * @param {number} plainHeight The height of the static chart
 50 		 * @return The current object
 51 		 * @type Sources
 52 		 */
 53 		add: function(key, abbr, url, src, flashVars, width, height, opacity,
 54 		              reTitle, reValue, reChange, reOthers,
 55 									plainOn, plainURL, plainWidth, plainHeight){
 56 		  sources[key] = {
 57 		    "url": url,
 58 		    embed: {
 59 		      "src": src,
 60 		      "flashVars": flashVars,
 61 		      "width": width,
 62 		      "height": height,
 63 		      "opacity": opacity
 64 		    },
 65 		    regex: {
 66 		      title: reTitle,
 67 		      value: reValue,
 68 		      change: reChange,
 69 					others: reOthers
 70 		    },
 71 				plain: {
 72 					on: (plainOn === true),
 73 					url: plainURL,
 74 					width: plainWidth,
 75 					height: plainHeight
 76 				},
 77 				abbr: abbr
 78 		  };
 79 			return this;
 80 		},
 81 		
 82 		/**
 83 		 * Exports the information in the Sources object in JSON format
 84 		 * @return A JSON representation of sources
 85 		 * @type object
 86 		 */
 87 		exportJSON: function(){
 88 			var temp = {};
 89 			$.each(sources, function(key, value){
 90 				temp[key] = {};
 91 				temp[key].url = value.url;
 92 				temp[key].embed = value.embed;
 93 				temp[key].plain = value.plain;
 94 			});
 95 			return temp;
 96 		},
 97 		
 98 		/**
 99 		 * 
100 		 * @param {string} label
101 		 * @return The information on a source
102 		 * @type object
103 		 */
104 		get: function(key){
105 		  return sources[key];
106 		},
107 		
108 		/**
109 		 * Creates an array containing which sources a metric is valid for,
110 		 * returning all sources if the metric cannot be found
111 		 * @return {string} An string of abbreviations for valid sources
112 		 */
113 		valid: function(metric){
114 			var found = '';
115 			var all = '';
116 			$.each(sources, function(key, source){
117 				found += (metric in source.regex.others ? source.abbr : ' ');
118 				all += source.abbr;
119 			});
120 
121 			// Regular expression tests for whether the found string contains nothing
122 			// but whitespace
123 			return (/^\s+$/.test(found) ? all : found);
124 		},
125 		
126 		/**
127 		 * Creates a group of radio buttons for selection
128 		 * @param {string} name A unique name for the radio buttons
129 		 * @param {string} currentSource The name of the current source for the
130 		 * quote 
131 		 * @return The context node
132 		 * @type object
133 		 */
134 		selection: function(name, currentSource){
135 			var $td = $('<td>');
136 			$.each(sources, function(key, value){
137 				$td.append($('<input>').val(key).attr({
138 					checked: (key === currentSource),
139 					id: name + '_' + key,
140 					name: name,
141 					title: Localizer.translate(key),
142 					type: 'radio'
143 				}));
144 			});
145 			return $td;
146 		}
147 	}
148 })();
